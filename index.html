-- OSS_v34-2v_offer_events.sql -- Gate for the Transaction Detail card (V8+ pillar #6 / provenance). -- WHY: the `offers` row holds only CURRENT state — every counter overwrites the -- last, so the negotiation history is being lost. This adds an append-only event -- log so a sold item can show its full offer -> counter -> ... -> accept timeline. -- -- Safe to run anytime. Client writes are best-effort (separate inserts), so the -- app keeps working whether or not this has been run yet. -- Idempotent: re-running is a no-op (IF NOT EXISTS + guarded backfill). -- 1) append-only event table create table if not exists public.offer_events ( id uuid primary key default gen_random_uuid(), offer_id uuid not null references public.offers(id) on delete cascade, actor_role text not null check (actor_role in ('buyer','seller')), actor_username text, event_type text not null check (event_type in ('offer','counter','accept','decline','withdraw','expire')), amount numeric(12,2), -- amount AT this step; null for decline/withdraw/expire currency text, -- 'USD' | 'SB' (the currency of THIS step) note text, -- optional (initial-offer message, etc.) created_at timestamptz not null default now() ); -- chronological read per offer create index if not exists offer_events_offer_id_created_idx on public.offer_events (offer_id, created_at); -- 2) RLS — only the two parties to the parent offer can read/write its events alter table public.offer_events enable row level security; -- read: buyer or seller on the parent offer (mirrors the offers SELECT policy) drop policy if exists offer_events_select_parties on public.offer_events; create policy offer_events_select_parties on public.offer_events for select using ( exists ( select 1 from public.offers o where o.id = offer_events.offer_id and (o.buyer_id = auth.uid() or o.seller_id = auth.uid()) ) ); -- insert: same party gate (the client appends its own action) drop policy if exists offer_events_insert_parties on public.offer_events; create policy offer_events_insert_parties on public.offer_events for insert with check ( exists ( select 1 from public.offers o where o.id = offer_events.offer_id and (o.buyer_id = auth.uid() or o.seller_id = auth.uid()) ) ); grant select, insert on public.offer_events to anon, authenticated; -- 3) backfill — seed one opening 'offer' event per existing offer -- so already-sold items show at least the opening bid in the timeline. insert into public.offer_events (offer_id, actor_role, actor_username, event_type, amount, currency, note, created_at) select o.id, 'buyer', o.buyer_username, 'offer', o.offer_amount, coalesce(o.buyer_offer_currency, o.offer_currency, o.listed_currency, 'SB'), o.message, o.created_at from public.offers o where not exists ( select 1 from public.offer_events e where e.offer_id = o.id );
